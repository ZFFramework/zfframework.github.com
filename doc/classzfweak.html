Detailed Description

weak reference to ZFObject

usage:

ZFArray *obj = xxx;
zfweak p1 = obj;
ZFArray *access1 = p1.get();
zfweakT<ZFArray> p2 = obj;
if(p2) {
p2.get()->add(xxx);
}
p2.set(xxx);
if(p2.valid()) {
p2.get()->add(xxx);
}
