156T_Element *zfmemmoveObject(T_Element *dst, const T_Element *src, zfindex count) {
157 if(dst && src && count != 0 && count != zfindexMax()) {
158 if(dst > src && src + count > dst) {
159 const T_Element *srcEnd = src;
160 src += count;
161 dst += count;
162 while(src > srcEnd) {
163 --dst;
164 --src;
165 *dst = *src;
166 }
167 }

169 const T_Element *srcEnd = src + count;
170 while(src < srcEnd) {
171 *dst = *src;
