152T_Element *zfmemmoveObject(T_Element *dst, const T_Element *src, zfindex count) {
153 if(dst && src && count != 0 && count != zfindexMax()) {
154 if(dst > src && src + count > dst) {
155 const T_Element *srcEnd = src;
156 src += count;
157 dst += count;
158 while(src > srcEnd) {
159 --dst;
160 --src;
161 *dst = *src;
162 }
163 }
164 else {
165 const T_Element *srcEnd = src + count;
166 while(src < srcEnd) {
167 *dst = *src;
168 ++dst;
169 ++src;
170 }
171 }
172 }
