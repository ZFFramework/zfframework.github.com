245 zfchar buf[64] = {0};
246 snprintf(buf, sizeof(buf), "%lf", (double)n);
247 const zfchar *p = buf;
248 while(*p && *p != '.') {++p;}
249 if(*p == '\0') {
250 s += buf;
251 return zftrue;
252 }
253 const zfchar *pEnd = p + zfslen(p) - 1;
254 if(*pEnd == '0') {
255 do {--pEnd;} while(*pEnd == '0');

261 }
262 }
