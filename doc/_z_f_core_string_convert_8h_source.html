125template<typename T_Int>
126zfbool zfsFromIntT(
127 ZF_OUT zfstring &s
128 , ZF_IN T_Int n
129 , ZF_IN_OPT zfindex radix = 10
130 , ZF_IN_OPT zfbool useUpperCase = zftrue
131 ) {
132 if(radix < 2 || radix > 36) {return zffalse;}
133
134 if(n <= (T_Int)0) {
135 // tricks to solve the unsigned type warnings
136 if(n != (T_Int)0) {
137 s += '-';
138 n = (T_Int)0 - n;
139 }
140 else {
141 s += '0';
142 return zftrue;
143 }
144 }
145 else if(n == (T_Int)-1) {
146 s += '-';
147 s += '1';
148 return zftrue;
149 }
150
151 unsigned long long e = 1;
152 while(n / e) {e *= radix;}
153
154 const zfchar *token = useUpperCase
155 ? "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ"
156 : "0123456789abcdefghijklmnopqrstuvwxyz";
157 while(e /= radix) {
158 s += token[n / e];
159 n %= e;
160 }
161 return zftrue;
162}

187zfbool zfsToIntT(
188 ZF_OUT T_Int &ret
189 , ZF_IN const zfchar *src
190 , ZF_IN_OPT zfindex srcLen = zfindexMax()
191 , ZF_IN_OPT zfindex radix = 10
192 , ZF_IN_OPT zfbool allowNegative = zftrue
193 , ZF_OUT_OPT const zfchar **outErrorPos = zfnull
194 ) {
195 if(src == zfnull || srcLen == 0 || radix < 2 || radix > 36) {
196 if(outErrorPos != zfnull) {
197 *outErrorPos = src;
198 }
199 return zffalse;
200 }
201
202 ret = 0;
203 const zfchar *p = src;
204 const zfchar *pEnd = ((srcLen == zfindexMax()) ? (p - 1) : (p + srcLen));
205 zfbool negative = zffalse;
206 if(*p == '-') {
207 if(!allowNegative) {
208 if(outErrorPos != zfnull) {
209 *outErrorPos = p;
210 }
211 return zffalse;
212 }
213 negative = zftrue;
214 ++p;
215 }
216 while(*p != '\0' && p != pEnd) {
217 zfuint tmp = 0;
218 if(*p >= '0' && *p <= '9') {tmp = *p - '0';}
219 else if(*p >= 'a' && *p <= 'z') {tmp = 10 + *p - 'a';}
220 else if(*p >= 'A' && *p <= 'Z') {tmp = 10 + *p - 'A';}
221 else {break;}
222 if(tmp >= radix) {break;}
223 ret = (T_Int)((ret * (zfuint)radix) + tmp);
224 ++p;
225 }
226 if(negative) {
227 ret = (T_Int)0 - ret;
228 }

276template<typename T_Float>
277zfbool zfsFromFloatT(
278 ZF_OUT zfstring &s
279 , ZF_IN T_Float n
280 ) {
281 zfchar buf[64] = {0};
282 snprintf(buf, sizeof(buf), "%lf", (double)n);
283 const zfchar *p = buf;
284 while(*p && *p != '.') {++p;}
285 if(*p == '\0') {
286 s += buf;
287 return zftrue;
288 }
289 const zfchar *pEnd = p + zfslen(p) - 1;
290 if(*pEnd == '0') {
291 do {--pEnd;} while(*pEnd == '0');
292 if(pEnd == p) {
293 s.append(buf, p - buf);
294 }
295 else {
296 s.append(buf, pEnd + 1 - buf);
297 }
298 }
299 else {
300 s += buf;
301 }
302 return zftrue;
303}
