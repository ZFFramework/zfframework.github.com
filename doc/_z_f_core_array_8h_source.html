890  d->contentEnd = newBuf + (oldContentEnd - oldBuf);
891 
892  _ZFP_ZFCoreArray_objDestroy(oldBuf, oldContentEnd, d->PODType);
893  zffree(oldBuf);
894  }
895  }
896  }
