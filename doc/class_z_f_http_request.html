Detailed Description

http request util

typical usage:

ZFLISTENER(onRecv) {
ZFHttpResponse *response = zfargs.param0();
ZFLogTrim() << response->body();
} ZFLISTENER_END()
zfobj<ZFHttpRequest>("http://xxx", "POST")
->header("Content-Type", "application/json;charset=UTF-8;")
->body(ZFJson()
.attr("k", "v")
.toString()
)
->start(onRecv);
